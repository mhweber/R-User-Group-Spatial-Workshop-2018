Interactive Maps

Make a leaflet map

library(leaflet)

m <- leaflet() %>%
  addTiles() %>%  # Add default OpenStreetMap map tiles
  addMarkers(lng=174.768, lat=-36.852, popup="The birthplace of R")
m  # Print the map

Make a plotly figure

library(plotly)
p <- plot_ly(midwest, x = ~percollege, color = ~state, type = "box")
